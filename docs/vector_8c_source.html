317void vector_spread(vector_t *const vector, const size_t index, const size_t amount)
318{
319 assert(vector);
320 assert(amount > 1);
321 assert((index < vector->capacity) && "Index out of capacity bounds!");
322 assert((index + amount <= vector->capacity) && "`index + amount` exceedes vector's capacity.");
323
324 size_t elements_set = 1;
325
326 /* copy elements exponentially through out the range */
327 for (; (elements_set + elements_set) <= amount; elements_set += elements_set)
328 {
329 void *dest = vector_get(vector, index + elements_set);
330 vector_copy(vector, dest, index, elements_set);
331 }
332 /* copy rest that left */
333 for (; elements_set < amount; ++elements_set)
334 {
335 void *dest = vector_get(vector, index + elements_set);
336 vector_copy(vector, dest, index + elements_set - 1, 1);
337 }
338}

465static void *binary_find(const vector_t *const vector,
466 const void *const value,
467 const size_t start,
468 const size_t end,
469 const compare_t cmp,
470 void *param)
471{
472 if (start == end)
473 {
474 return NULL;
475 }
476
477 const size_t middle = (start + end) / 2;
478 void *middle_value = vector_get(vector, middle);
479
480 if (0 == cmp(value, middle_value, param))
481 {
482 return middle_value;
483 }
484
485 if (0 < cmp(value, middle_value, param))
486 {
487 return binary_find(vector, value, middle + 1, end, cmp, param);
488 }
489
490 return binary_find(vector, value, start, middle, cmp, param);
491}
