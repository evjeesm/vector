311 void vector_spread(vector_t *const vector, const size_t index, const size_t amount)
312 {
313  assert(vector);
314  assert(amount > 1);
315  assert((index < vector->capacity) && "Index out of capacity bounds!");
316  assert((index + amount <= vector->capacity) && "`index + amount` exceedes vector's capacity.");
317 
318  size_t elements_set = 1;
319 
320  /* copy elements exponentially through out the range */
321  for (; (elements_set + elements_set) <= amount; elements_set += elements_set)
322  {
323  void *dest = vector_get(vector, index + elements_set);
324  vector_copy(vector, dest, index, elements_set);
325  }
326  /* copy rest that left */
327  for (; elements_set < amount; ++elements_set)
328  {
329  void *dest = vector_get(vector, index + elements_set);
330  vector_copy(vector, dest, index + elements_set - 1, 1);
331  }
